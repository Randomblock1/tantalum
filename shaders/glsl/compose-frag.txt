#include "preamble"

uniform sampler2D Frame;
uniform sampler2D PreviewFrame;
uniform float Exposure;
uniform float PreviewMix;

in vec2 vTexCoord;

out vec4 outColor;

void main() {
    vec3 rgb = texture(Frame, vTexCoord).rgb + texture(PreviewFrame, vTexCoord).rgb * PreviewMix;
    outColor = vec4(pow(rgb * Exposure, vec3(1.0 / 2.2)), 1.0);
}
